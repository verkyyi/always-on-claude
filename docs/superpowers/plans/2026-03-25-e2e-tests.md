# E2E Test Suite Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a plain-bash e2e test suite for all runtime scripts, runnable locally and in CI.

**Architecture:** Per-script test files source a shared `tests/test-lib.sh` that provides a runner, assertions, and fixtures. `tests/run.sh` is the entry point. Each test gets an isolated `$TEST_DIR` with fake `$HOME`. External tools (docker, tmux, nproc) are mocked via PATH-prepended stubs; git is used for real.

**Tech Stack:** Bash, git (real), mock binaries for docker/tmux/nproc/awk

### Task 1: Test library — runner, assertions, fixtures

**Files:**

This is the foundation everything else depends on. It provides the test runner, assertion functions, and fixture helpers.

- [ ] **Step 1: Create `tests/test-lib.sh` with runner and assertions**

```bash
#!/bin/bash
# test-lib.sh — Shared test runner, assertions, and fixtures.
# Sourced by each test-*.sh file.

set -euo pipefail

_PASS=0
_FAIL=0
_CURRENT_TEST=""

# --- Colors ---
_RED=$'\033[0;31m'
_GREEN=$'\033[0;32m'
_RESET=$'\033[0m'

# --- Runner ---

# Called before each test. Override in test files for custom setup.
setup() { :; }

# Called after each test. Override in test files for custom teardown.
teardown() { :; }

# Run all test_* functions in the calling script
run_tests() {
    local test_funcs
    test_funcs=$(declare -F | awk '/test_/ {print $3}' | sort)

    for func in $test_funcs; do
        _CURRENT_TEST="$func"
        TEST_DIR=$(mktemp -d)
        mkdir -p "$TEST_DIR/home" "$TEST_DIR/bin" "$TEST_DIR/markers"
        export HOME="$TEST_DIR/home"
        export PATH="$TEST_DIR/bin:$_ORIG_PATH"

        local failed=false
        setup
        if ! "$func"; then
            failed=true
        fi
        teardown
        rm -rf "$TEST_DIR"

        if [[ "$failed" == true ]]; then
            echo "  ${_RED}FAIL${_RESET} $func"
            ((_FAIL++))
        else
            echo "  ${_GREEN}PASS${_RESET} $func"
            ((_PASS++))
        fi
    done
}

print_summary() {
    local total=$((_PASS + _FAIL))
    echo ""
    if [[ $_FAIL -eq 0 ]]; then
        echo "${_GREEN}All $total tests passed${_RESET}"
    else
        echo "${_RED}$_FAIL/$total tests failed${_RESET}"
    fi
    return $_FAIL
}

# Save original PATH before any mocks
_ORIG_PATH="$PATH"

# --- Assertions ---

_fail() {
    local msg="${1:-}"
    echo "    ASSERT FAILED${msg:+: $msg}" >&2
    return 1
}

assert_eq() {
    local expected="$1" actual="$2" msg="${3:-expected '$1' got '$2'}"
    [[ "$expected" == "$actual" ]] || _fail "$msg"
}

assert_neq() {
    local a="$1" b="$2" msg="${3:-expected '$1' != '$2'}"
    [[ "$a" != "$b" ]] || _fail "$msg"
}

assert_contains() {
    local haystack="$1" needle="$2" msg="${3:-'$1' does not contain '$2'}"
    [[ "$haystack" == *"$needle"* ]] || _fail "$msg"
}

assert_file_exists() {
    local path="$1"
    [[ -e "$path" ]] || _fail "file does not exist: $path"
}

assert_file_not_exists() {
    local path="$1"
    [[ ! -e "$path" ]] || _fail "file should not exist: $path"
}

assert_exit_code() {
    local expected="$1"; shift
    local actual=0
    "$@" >/dev/null 2>&1 || actual=$?
    assert_eq "$expected" "$actual" "exit code: expected $expected got $actual"
}

assert_output_contains() {
    local needle="$1"; shift
    local output
    output=$("$@" 2>&1) || true
    assert_contains "$output" "$needle" "output does not contain '$needle'"
}

# --- Fixtures ---

create_test_repo() {
    local name="${1:-test-repo}"
    local repo="$HOME/$name"
    git init -q "$repo"
    git -C "$repo" config user.name "Test"
    git -C "$repo" config user.email "test@test.com"
    touch "$repo/README.md"
    git -C "$repo" add .
    git -C "$repo" commit -q -m "Initial commit"
    echo "$repo"
}

create_test_worktree() {
    local repo="$1" branch="$2"
    git -C "$repo" worktree add -q -b "$branch" "${repo}--${branch}" 2>/dev/null
    echo "${repo}--${branch}"
}

mock_binary() {
    local name="$1" output="${2:-}"
    local mock="$TEST_DIR/bin/$name"
    cat > "$mock" <<MOCK
#!/bin/bash
echo "\$0 \$*" >> "$TEST_DIR/bin/${name}.log"
echo "$output"
MOCK
    chmod +x "$mock"
}

# Mock binary that writes a marker file (for exec detection)
mock_binary_marker() {
    local name="$1" marker_name="${2:-$1}"
    local mock="$TEST_DIR/bin/$name"
    cat > "$mock" <<MOCK
#!/bin/bash
echo "\$0 \$*" > "$TEST_DIR/markers/${marker_name}"
MOCK
    chmod +x "$mock"
}
```

```bash
#!/bin/bash
# run.sh — Run all test-*.sh files, or specific files passed as args.
#
# Usage:
#   bash tests/run.sh                          # run all
#   bash tests/run.sh tests/test-update.sh     # run one file

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
REPO_ROOT="$(cd "$SCRIPT_DIR/.." && pwd)"

cd "$REPO_ROOT"

total_pass=0
total_fail=0
files=("${@}")

if [[ ${#files[@]} -eq 0 ]]; then
    mapfile -t files < <(find tests -name 'test-*.sh' -type f | sort)
fi

for file in "${files[@]}"; do
    echo ""
    echo "=== $file ==="
    (
        source tests/test-lib.sh
        source "$file"
        run_tests
        print_summary
    )
    result=$?
    if [[ $result -ne 0 ]]; then
        total_fail=1
    fi
done

echo ""
if [[ $total_fail -ne 0 ]]; then
    echo "Some test files had failures."
    exit 1
else
    echo "All test files passed."
    exit 0
fi
```

```bash
# --- cleanup ---

test_cleanup_dry_run() {
    local repo
    repo=$(create_test_repo projects/myrepo)
    create_test_worktree "$repo" "merged-branch"

    # Merge the branch into main so cleanup considers it merged
    git -C "$repo" merge --no-edit "$HOME/projects/myrepo--merged-branch" 2>/dev/null || true
    # Push to a fake origin so merge-base --is-ancestor works
    local bare="$TEST_DIR/bare-remote"
    git init -q --bare "$bare"
    git -C "$repo" remote add origin "$bare" 2>/dev/null || true
    git -C "$repo" push -q origin main 2>/dev/null || true

    local output
    output=$(bash "$HELPER" cleanup --dry-run)
    assert_contains "$output" "dry-run"
    # Worktree should still exist
    assert_file_exists "$HOME/projects/myrepo--merged-branch/.git"
}

test_cleanup_removes_merged() {
    local repo
    repo=$(create_test_repo projects/myrepo)
    create_test_worktree "$repo" "merged-branch"

    # Make a commit on the worktree branch
    touch "$HOME/projects/myrepo--merged-branch/newfile"
    git -C "$HOME/projects/myrepo--merged-branch" add .
    git -C "$HOME/projects/myrepo--merged-branch" commit -q -m "worktree commit"

    # Merge into main
    git -C "$repo" merge --no-edit merged-branch 2>/dev/null || true

    # Set up origin
    local bare="$TEST_DIR/bare-remote"
    git init -q --bare "$bare"
    git -C "$repo" remote add origin "$bare" 2>/dev/null || true
    git -C "$repo" push -q origin main 2>/dev/null || true

    assert_file_not_exists "$HOME/projects/myrepo--merged-branch"
}
```

```bash
#!/bin/bash
# Tests for scripts/runtime/ssh-login.sh

LOGIN_SCRIPT="$REPO_ROOT/scripts/runtime/ssh-login.sh"

setup() {
    mkdir -p "$HOME/dev-env/scripts/runtime"
}

# Helper: run ssh-login.sh in a subshell simulating interactive SSH
# Returns stdout; sets _EXIT_CODE
run_login() {
    local extra_env="${1:-}"
    _EXIT_CODE=0
    # bash -i makes it interactive ($- contains 'i')
    # We source the script so 'return' works (not 'exit')
    _OUTPUT=$(
        env HOME="$HOME" SSH_CONNECTION="1.2.3.4 1234 5.6.7.8 22" \
            $extra_env \
            bash -i -c "source '$LOGIN_SCRIPT'" 2>&1
    ) || _EXIT_CODE=$?
}

test_skips_non_interactive() {
    # Non-interactive shell: bash -c (without -i) has $- without 'i'
    # ssh-login.sh uses 'return' which exits the sourcing process with non-zero
    _EXIT_CODE=0
    _OUTPUT=$(
        env HOME="$HOME" SSH_CONNECTION="1.2.3.4 1234 5.6.7.8 22" \
            bash -c "source '$LOGIN_SCRIPT'" 2>&1
    ) || _EXIT_CODE=$?
    # Should not attempt to exec anything
    assert_file_not_exists "$TEST_DIR/markers/start-claude"
    assert_file_not_exists "$TEST_DIR/markers/onboarding"
}
```

### Task 5: start-claude.sh tests

**Files:**
- Create: `tests/test-start-claude.sh`
- Reads: `scripts/runtime/start-claude.sh`

The challenge: `start-claude.sh` has top-level code that runs on source (container check, `discover` call, main loop). We can't simply source it. Instead, we extract and test individual functions by grep-sourcing them from the script, or by running the full script with mocked binaries that make top-level code no-op quickly.

Strategy: extract function definitions from the script and source only those, avoiding top-level execution.

- [ ] **Step 1: Create `tests/test-start-claude.sh`**

```bash
#!/bin/bash
# Tests for scripts/runtime/start-claude.sh

START_SCRIPT="$REPO_ROOT/scripts/runtime/start-claude.sh"

# Extract function definitions from start-claude.sh without running top-level code.
# We source only the function blocks and variable definitions.
_source_functions() {
    # Set variables the functions depend on
    export COMPOSE_DIR="$HOME/dev-env"
    export CONTAINER_NAME="claude-dev"
    export WORKTREE_HELPER="$COMPOSE_DIR/scripts/runtime/worktree-helper.sh"
    export CONTAINER_PROJECTS="/home/dev/projects"

    # Extract function definitions using sed: from "^func_name()" to "^}"
    eval "$(sed -n '/^count_sessions()/,/^}/p' "$START_SCRIPT")"
    eval "$(sed -n '/^get_max_sessions()/,/^}/p' "$START_SCRIPT")"
    eval "$(sed -n '/^check_session_limit()/,/^}/p' "$START_SCRIPT")"
    eval "$(sed -n '/^to_container_path()/,/^}/p' "$START_SCRIPT")"
    eval "$(sed -n '/^discover()/,/^}/p' "$START_SCRIPT")"
}

setup() {
    mkdir -p "$HOME/dev-env/scripts/runtime" "$HOME/projects"
    # Default mock: tmux returns no sessions
    mock_binary tmux ""
    # Default mock: nproc returns 2
    mock_binary nproc "2"
    _source_functions
}

MOCK

test_max_sessions_env_override() {

test_max_sessions_calculated() {
    # 4096 MB, 2 CPUs -> mem_based = (4096-1024)/650 = 4, min(4,2) = 2
    mock_binary nproc "2"
    cat > "$TEST_DIR/bin/awk" <<'MOCK'
#!/bin/bash
if [[ "$*" == *"MemTotal"* ]]; then
    echo "4096"
else
    /usr/bin/awk "$@"
fi
MOCK
    chmod +x "$TEST_DIR/bin/awk"
    _source_functions
```
